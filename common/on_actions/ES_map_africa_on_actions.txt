on_actions = {
	on_startup = {
		effect = {
			BOR = {	
				init_map_africa_global_data = yes
			}	

			WGN = { 
				set_variable = { map_africa_selected_region = -1 }
				map_africa_calculate_RUS_all_regions_recommended_pool_value = yes
			}
			NTO = { 
				set_variable = { map_africa_selected_region = -1 }
				map_africa_calculate_NTO_all_regions_recommended_pool_value = yes
			}
		}
	}

	on_weekly = {
		effect = {
			if = {
				limit = { has_country_flag = map_africa }

				if = {
					limit = { OR = {
						original_tag = RUS
						original_tag = WGN
					}}

					map_africa_calculate_RUS_all_regions_recommended_pool_value = yes
				}
				else_if = {
					limit = { original_tag = NTO }
					
					map_africa_calculate_NTO_all_regions_recommended_pool_value = yes
				}

				map_africa_tick = yes
				map_africa_tick_ai_regions_manpower_pools = yes
			}
			
		}
	}
}
