on_actions = {

    # Показывается при запуске новой игры
    on_game_start = {
        effect = {
            # Показываем только игроку, не ИИ
            every_country = {
                limit = { is_ai = no }

                # Инициализация переменных (чтобы кнопки не глючили)
                set_variable = { var = show_up value = 0 }
                set_variable = { var = show_up_info value = 0 }
                set_variable = { var = show_up_unit value = 0 }
                set_variable = { var = show_up_money value = 0 }
                set_variable = { var = show_up_credit value = 0 }
                set_variable = { var = show_up_country value = 0 }

                # Открываем окно меню (имя корневого окна из .gui)
                open_gui = startup_menu_
            }
        }
    }

    # Можно также открыть меню при загрузке сейва, если хочешь
    on_load = {
        effect = {
            every_country = {
                limit = { is_ai = no }
                if = {
                    limit = { has_variable = esd_menu_reopen }
                    open_gui = startup_menu_
                }
            }
        }
    }
}
