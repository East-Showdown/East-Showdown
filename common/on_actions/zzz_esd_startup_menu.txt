on_actions = {

    # Показывается при запуске новой игры
    on_startup = {
        effect = {
            # Показываем только игроку, не ИИ
            every_country = {
                limit = { is_ai = no }

                # Инициализация переменных (чтобы кнопки не глючили) цифорку ставил чтобы открывалась 1 вкладка
                set_variable = { var = show_up value = 0 }
                set_variable = { var = show_up_info value = 0 }
                set_variable = { var = show_up_unit value = 0 }
                set_variable = { var = show_up_money value = 0 }
                set_variable = { var = show_up_credit value = 0 }
                set_variable = { var = show_up_country value = 0 }
                set_variable = { var = show_up_ammo value = 0 }
            }
        }
    }

}
