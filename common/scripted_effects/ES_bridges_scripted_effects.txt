weekly_hit_KyivHPP = { 
	set_temp_variable = { weekly_damage = global.Kyiv_HPP_MAX_HP }
	multiply_temp_variable = { var = weekly_damage value = global.DAM_WEEKLY_HP_LOSS_PERCENT}

	if = { limit = { check_variable = { global.Kyiv_HPP_CURRENT_HP < global.Kyiv_HPP_START_LOSS_HP  } } 
		subtract_from_variable = {
			global.Kyiv_HPP_CURRENT_HP = weekly_damage
		}
	}
}
