#Update All Dnipro Objects
update_all_dnipro_objects = {

	log = "INFO:update_all_dnipro_objects: entering temp_obj_name for-each loop"



    for_each_loop = {
        array = global.dnipro_objects
        value = temp_obj_name
        index = i

        clamp_variable = {
            var = global.dnipro_obj_current_hp^i
            min = 0
            max = global.dnipro_obj_max_hp^i
        }


        if = {
            limit = {
                check_variable = { global.dnipro_obj_is_dam^i = 1 }
                check_variable = { global.dnipro_obj_current_hp^i < global.dnipro_obj_start_loss_hp^i }
            }

			subtract_from_variable = { global.dnipro_obj_current_hp^i = global.dnipro_obj_damage^i }
        }

        clamp_variable = {
            var = global.dnipro_obj_current_hp^i
            min = 0
            max = global.dnipro_obj_max_hp^i
        }
	}
	log = "INFO:update_all_dnipro_objects: temp_obj_name - exit"
	dnipro_sync_controllers_with_provinces = yes	
	RUS = { autorepair_all_dnipro_objects = yes }
	UKR = { autorepair_all_dnipro_objects = yes }
	log = "INFO:update_all_dnipro_objects: under - exit"	
	autoblocking_all_dnipro_straits_objects = yes
}

autorepair_all_dnipro_objects = { 
	log = "INFO:update_all_dnipro_objects: var THIS.bridges_repairing_is_allowed: [?THIS.GetTag] [?THIS.bridges_repairing_is_allowed]"
	
	if = {
			limit = { 
				check_variable = { THIS.bridges_repairing_is_allowed = 1 }
			}
			set_temp_variable = { THIS.repair_rate = 69 }	
			log = "INFO:update_all_dnipro_objects: temp repair_rate initialized: [?THIS.repair_rate]"

			set_temp_variable = { repair_for_code = 1 }  
			if = { limit = { original_tag = RUS } set_temp_variable = { repair_for_code = 2 } }

			if = { 
				limit = { 
					check_variable = { THIS.REPAIR_PRIORITY = 1 }
				}
			set_temp_variable = { THIS.repair_rate = global.DEFAULT_AUTO_REPAIR_RATE_LOW_PRIORITY }
			log = "INFO:update_all_dnipro_objects: value of temp repair_rate setted: [?THIS.repair_rate]"
			}

			if = { 
				limit = { 
					check_variable = { THIS.REPAIR_PRIORITY = 2 }
				}
			set_temp_variable = { THIS.repair_rate = global.DEFAULT_AUTO_REPAIR_RATE_MEDIUM_PRIORITY }
			log = "INFO:update_all_dnipro_objects: value of temp repair_rate setted: [?THIS.repair_rate]"
			}
			
			if = { 
				limit = { 
					check_variable = { THIS.REPAIR_PRIORITY = 3 }
				}
			set_temp_variable = { THIS.repair_rate = global.DEFAULT_AUTO_REPAIR_RATE_HIGH_PRIORITY }
			log = "INFO:update_all_dnipro_objects: value of temp repair_rate setted: [?THIS.repair_rate]"
			}
			
			log = "INFO:update_all_dnipro_objects: entering under for-each loop"
			for_each_loop = {
				array = global.dnipro_objects
				index = i
				value = _under	
				
			if = {
				limit = {
					check_variable = { global.bridges_province_controllers^i = repair_for_code }
					check_variable = { global.dnipro_obj_current_hp^i < global.dnipro_obj_max_hp^i }
				}

				set_temp_variable = { THIS.repair_add = global.dnipro_obj_max_hp^i }
				log = "INFO:update_all_dnipro_objects: temp THIS.repair_add initialized: [?THIS.repair_add]"
				multiply_temp_variable = { THIS.repair_add = repair_rate }      
				log = "INFO:update_all_dnipro_objects: temp THIS.repair_add * temp repair_rate: [?THIS.repair_add]"

				add_to_variable = { var = global.dnipro_obj_current_hp^i value = repair_add }
				log = "INFO:update_all_dnipro_objects: temp THIS.repair_add * temp repair_rate: [?THIS.repair_add]"
				clamp_variable = {
					var = global.dnipro_obj_current_hp^i
					min = 0
					max = global.dnipro_obj_max_hp^i
				}								
			}
		}
	}
}
