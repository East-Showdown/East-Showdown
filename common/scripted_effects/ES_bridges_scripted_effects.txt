#Update All Dnipro Objects
update_all_dnipro_objects = {

    for_each_loop = {
        array = global.dnipro_objects
        value = temp_obj_name
        index = i


        set_temp_variable = { is_dam = global.dnipro_obj_is_dam^i }

        set_temp_variable = { cur_hp = global.dnipro_obj_current_hp^i }
        set_temp_variable = { max_hp = global.dnipro_obj_max_hp^i }
        set_temp_variable = { strait_id = global.dnipro_obj_strait^i }
        set_temp_variable = { start_loss_hp = global.dnipro_obj_start_loss_hp^i }
        set_temp_variable = { weekly_loss = global.dnipro_obj_weekly_loss^i }
        set_temp_variable = { crit_hp = global.dnipro_obj_critical_hp^i }
        set_temp_variable = { repair_hp = global.dnipro_obj_repair_hp^i }


        clamp_variable = {
            var = cur_hp
            min = 0
            max = var:max_hp
        }

        set_variable = {
            var = dnipro_obj_current_hp^i
            value = var:cur_hp
        }

        if = {
            limit = {
                check_variable = { var:is_dam = 1 }
                check_variable = { var:cur_hp < var:start_loss_hp }
            }

            set_temp_variable = { damage = var:max_hp }
            multiply_temp_variable = { var = damage value = var:weekly_loss }
            subtract_from_temp_variable = { var = cur_hp value = var:damage }

        }

        clamp_variable = {
            var = cur_hp
            min = 0
            max = var:max_hp
        }

        set_variable = {
            var = dnipro_obj_current_hp^i
            value = var:cur_hp
        }

        if = {
            limit = {
                check_variable = { var:cur_hp <= 0 }
            }
            set_global_flag = var:strait_id
            else = { set_temp_variable = { skip_rest = 1 } }
        }

        if = {
            limit = {
                check_variable = { var:skip_rest < 1 }  
                check_variable = { var:cur_hp < var:crit_hp }
                check_variable = { var:cur_hp > 0 }
            }
            set_global_flag = var:strait_id
        }

        if = {
            limit = {
                check_variable = { var:skip_rest < 1 }
                check_variable = { var:cur_hp >= var:repair_hp }
            }
            clr_global_flag = var:strait_id
        }

    }
}
