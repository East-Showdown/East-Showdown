						set_temp_variable = { ROOT.ammunitionSilo = THIS.non_damaged_building_level@ammunition_silo }
					}
				}
				visual_effect = {
					entity = "raid_failed_entity"
					animation = "attack"
				}
			}
	
			limited_success = {
				victim_effects = {
					var:target_state = {
						set_temp_variable = { ROOT.ammunitionSilo = THIS.non_damaged_building_level@ammunition_silo }

						var = building_damage_by_missile

						max = max_clamp

								text = {

					entity = "air_raids_2_limited_success_entity"
