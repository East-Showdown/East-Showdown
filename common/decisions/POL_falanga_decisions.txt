	POL_counterrevolution_mission = {
		icon = generic_army_support

		allowed = { always = no }
		
		available = {
			country_exists = POL
			NOT = { has_completed_focus = POL_ban_communism }
			NOT = {
			check_variable = {
				FALANGA_COMMUNISM_CELLS = 13
			}
			}
			hidden_trigger = { 
				always = no 
				has_country_flag = POL_counterrevolution_flag
			}
		}

		cancel_trigger = {
			has_completed_focus = POL_ban_communism
			check_variable = {
				FALANGA_COMMUNISM_CELLS = 13
			}
		}

		visible = {
			country_exists = POL
			hidden_trigger = { 
				has_country_flag = POL_counterrevolution_flag
			}
		}

		days_mission_timeout = 120

		is_good = no
