	retake_anarchy_lands_decision = {

        state_target = yes
		on_map_mode = map_only
        target_trigger = {
            FROM = {
				NOT = {
					is_core_of = G4P
				}
				is_owned_by = G4P
				any_neighbor_state = { is_owned_by = ROOT }
				NOT = {
					is_core_of = ROOT
					is_claimed_by = ROOT
				}	
            }
        }
		priority = 50

		icon = GFX_decision_generic_form_nation

		ai_will_do = {
			base = 1
			modifier = {
				add = 1
				has_political_power > 50
			}
			modifier = {
				factor = 10
				has_political_power > 250
			}
			modifier = {
				factor = 10
				has_political_power > 400
			}
			modifier = {
				factor = -50
				any_controlled_state = {
					any_neighbor_state = {
						is_owned_by = G4P
						OR = {
							is_core_of = ROOT
							is_claimed_by = ROOT
						}
					}	
				}	
			}
		}

		available = {
			has_manpower > 8000
			FROM = {
				custom_trigger_tooltip = {
					tooltip = has_radiation_less_1
					check_variable = {
						var = radiation_var
						value = 1.0
						compare = less_than
					}
				}	
			}	
		}

		days_remove = 70

		days_re_enable = 10
		
		cancel_trigger = {
			custom_trigger_tooltip = {
				tooltip = has_anarchy_in_this_state_tt		
				FROM = {
					NOT = { is_owned_by = G4P }
				}
			}	
		}

		cost = 100

		complete_effect = {
			add_manpower = -3000
		}

		modifier = {
			political_power_gain = -0.5
			weekly_manpower = -250
		}

		remove_effect = {
			FROM = {
				transfer_state_to = ROOT
				set_demilitarized_zone = no
				add_manpower = 5000
				if = {
					limit = {
						state_population < 5000
					}
					add_core_of = ROOT
				}
			}	
		}
	}
